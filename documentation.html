Basic Syntax

FlowScript is designed to be a simple and intuitive scripting language. Here are some basic elements:

  • Variables: Define variables using let.
  • Functions: Define functions using func.
  • Control Structures: Includes if, while, and for loops.

Variables

Variables are declared using let. Example:

let x = 5

Functions

Functions are declared using func. Example:

func add(a, b) {
    return a + b
}
print(add(5, 3))

Control Structures

FlowScript supports standard control structures. Examples:

Conditional Statements


if x > 0 {
    print("Positive")
} else {
    print("Non-positive")
}

Loops

For Loop:

for i in range(0, 5) {
    print(i)
}

While Loop:

let i = 0
while i < 5 {
    print(i)
    i = i + 1
}

Examples

Rock, Paper, Scissors Game

An example of a simple game implemented in FlowScript:

func main() {
    print("Welcome to Rock, Paper, Scissors!")
    
    let playerChoice = getPlayerChoice()
    print(getChoiceName(playerChoice))

    let choices = [0, 1, 2]
    let random = Random().nextInt(0, 2)
    let computerChoice = choices[random]
    print(getChoiceName(computerChoice))

    if playerChoice == computerChoice {
        System().println("It's a tie!")
    } else if playerChoice == 1 and computerChoice == 2 {
        System().println("You win!")
    } else if playerChoice == 0 and computerChoice == 1 {
        System().println("You win!")
    } else if playerChoice == 2 and computerChoice == 0 {
        System().println("You win!")
    } else {
        System().println("Computer wins!")
    }
}
                
func getPlayerChoice() {
    while true {
        print("Choose one of the options: 0 - paper, 1 - rock, 2 - scissors")
        let choice = IO().readLine(">>> ")
        if choice == 0 {
            return choice
        } else if choice == 1 {
            return choice
        } else if choice == 2 {
            return choice
        } else {
            print("Invalid choice. Try again.")
        }
    }
}

func getChoiceName(choice) {
    if choice == 0 {
        return "paper"
    } else if choice == 1 {
        return "rock"
    } else {
        return "scissors"
    }
}
                
call main()

Math Functions

FlowScript provides several mathematical functions:

let x = 2.5

print(Math().abs(x))    // Absolute value
print(Math().sqrt(x))   // Square root
print(Math().pow(x, 3)) // Power
print(Math().sin(x))    // Sine
print(Math().cos(x))    // Cosine
print(Math().tan(x))    // Tangent
print(Math().getPI())   // Value of PI
